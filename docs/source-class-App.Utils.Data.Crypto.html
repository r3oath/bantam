 48:     * of the data is not n * blocksize, the data will be padded with '\0'.
 49:     * @param string $string The string to encrypt.
 50:     * @return string The base64 encoded encrypted string.

 52:     public static function encrypt($string, $key=null) {
 53:         if($key === null) {
 54:             $key = Config::get('bantam_app_secret_key', null);
 55:             if($key === null && !Session::hasBegun()) {
 56:                 throw new \ErrorException(self::key_error);
 57:             }

 59:             if(Session::hasBegun() && Session::valid()) {
 60:                 $key = Session::get('key');
 61:                 if($key === null) {
 62:                     $key = openssl_random_pseudo_bytes(self::key_bytes_size);
 63:                     Session::set('key', $key);
 64:                 }
 65:             }
 66:         }

106:     /**
107:     * Sign the given string and return the same string with the signature
108:     * appended with a '--' seperator. This function requires you either have
109:     * a unique, application specific 'key' set in your Config, or that you have
110:     * started a Session before this function is called. If neither of these
111:     * conditions are met, an ErrorException will be thrown.
112:     * @param string $string The string to sign.
113:     * @return string The signed string.
114:     */
115:     public static function sign($string) {
116:         $salt = Config::get('bantam_app_secret_key', null);
117:         if($salt === null && !Session::hasBegun()) {
118:             throw new \ErrorException(static::salt_error);
119:         }
120: 
121:         if(Session::hasBegun() && Session::isValid()) {
122:             $salt = Session::get('key');
123:             if($salt === null) {
124:                 $salt = openssl_random_pseudo_bytes(self::key_bytes_size);
125:                 Session::set('key', $salt);
126:             }
127:         }

130:         return $string.'--'.$checksum;
