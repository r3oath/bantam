 30: /**
 31: * Provides basic URL routing and URL token extraction.
 32: * <code>
 33: * // Register a new GET request for users/:name, grabbing
 34: * // the name provided and printing it out.
 35: *
 36: * Route::get('users/:name', function(){
 37: *     $name = Route::getToken('name');
 38: *     echo "Hello {$name}";
 39: * });
 40: * </code>
 41: * @package App\Utils\Http
 42: * @author Tristan Strathearn <r3oath@gmail.com>
 43: * @license http://opensource.org/licenses/MIT MIT License (MIT)
 44: */
 45: class Route {
 46:     /**
 47:     * @var array $routes A table of all registered routes and their properties.
 48:     */
 49:     private static $routes = array();
 50: 
 51:     /**
 52:     * @var array $route_tokens An array of all captured route tokens and thier values.
 53:     */
 54:     private static $route_tokens = array();
 55: 
 56:     /**
 57:     * Registers a new route internally.
 58:     * @param string $req_type The type of request this route handles.
 59:     * @param string $signature The signature of this route.
 60:     * @param mixed $closure The callback function for this route.
 61:     * @param array $regexs An array of tokens and their corresponding regexs to match against.
 62:     * @return void.
 63:     */
 64:     private static function register($req_type, $signature, $closure, $regexs) {
 65:         if(!isset(static::$routes[$req_type])) {
 66:             $routes[$req_type] = array();
 67:         }
 68: 
 69:         static::$routes[$req_type][] = array(
 70:             'signature' => $signature,
 71:             'closure' => $closure,
 72:             'regexs' => $regexs,
 73:         );
 74:     }

 77:     * Registers a new route that will handle a POST request.

 79:     * @param mixed $closure The callback function for this route.

120:     /**
121:     * Handles the given request by finding any matching routes in the routing table and executing
122:     * their callback functions.
123:     * @param array $request The request array, usually a product of Request::capture().
124:     * @return mixed True if the request was handled by a route, false otherwise.
125:     */
126:     public static function handle($request) {
127:         // Make sure there is at least 1 route type to handle this request.
128:         if(!isset(static::$routes[Request::getType()])) {
129:             return false;
130:         }
131: 
132:         // Get only the routes that handle this request type.
133:         $routes = static::$routes[Request::getType()];
134: 
135:         // Iterate over every entry in our routing table to find a possible
136:         // match to the given request.
137:         foreach ($routes as $route) {
138:             // Check to see if the request and signature have the same number
139:             // of constants and tokens, then proceed.
140:             $sig_parts = explode('/', $route['signature']);
141:             if(count($request) === count($sig_parts)) {
142:                 // Check each signature constant matches the contant given
143:                 // in the route and record any token values.
144:                 $matches = 0;
145:                 for ($i = 0; $i < count($request); $i++) {
146:                     if($sig_parts[$i] == $request[$i]) {
147:                         $matches += 1;
148:                     }
149:                     // If we've got ourselves a token here, remove the prepending :
150:                     // marker and add it's corresponding value from the request into
151:                     // the token array to be used later by the user.
152:                     if(strstr($sig_parts[$i], ':') !== false) {
153:                         static::$route_tokens[substr($sig_parts[$i], 1)] = $request[$i];
154:                         $matches += 1;
155:                     }
156:                 }
157: 
158:                 // If we found the right number of constants and tokens, we have
159:                 // found a matching route.
160:                 if($matches === count($sig_parts)) {
161: 
162:                     // Check to make sure that each token matches it's given regex (if provided).
163:                     $failure = false;
164:                     if(isset($route['regexs']) && count($route['regexs']) > 0) {
165:                         // Step through each of the token regex entries.
166:                         foreach ($route['regexs'] as $key => $value) {
167:                             // Check to see if this token has a corresponding regex.
168:                             if(array_key_exists($key, static::$route_tokens)) {
169:                                 // Match the token value against the regex.
170:                                 $result = preg_match("/\A{$value}\Z/", static::$route_tokens[$key]);
171:                                 // Mark failure on failure.
172:                                 if($result === false || $result === 0) {
173:                                     $failure = true;
174:                                 }
175:                             }
176:                         }
177:                     }
178: 
179:                     // Fail if the tokens did not match their respective regexs.
180:                     if($failure === true) {
181:                         static::$route_tokens = array();
182:                         return false;
183:                     }
184: 
185:                     // Run the user specified code here.
186:                     ob_start();
187:                     if($route['closure'] instanceOf Closure) {
188:                         $route['closure']();
189:                     } else if(is_callable($route['closure'])) {
190:                         call_user_func($route['closure']);
191:                     } else {
192:                         throw new \ErrorException('Unknown type of callback function for route.');
193:                     }
194:                     ob_end_flush();
195: 
196:                     // Return success.
197:                     return true;
198:                 }
199:             }
200:         }
201: 
202:         // No matching routes were found for the given request.
203:         static::$route_tokens = array();
204:         return false;
205:     }
206: 
207:     /**
208:     * Returns the available captured tokens for this request.
209:     * @return array The available tokens.
210:     */
211:     public static function getTokens() {
212:         return static::$route_tokens;
213:     }
214:

216:     * Find and return the specified token from the available captured tokens for this request.
217:     * @param string $key The specific token to look for.
218:     * @return mixed The string token if available, otherwise null.
219:     */
220:     public static function getToken($key) {
221:         if(isset(static::$route_tokens[$key])) {
222:             return static::$route_tokens[$key];
223:         }
224: 
225:         return null;
226:     }
